export const getProp = (props, name, defaultValue = '') => {
  return props[name] || defaultValue;
};

| Attribute        | Type     | Description                                                                                           |
|------------------|----------|-------------------------------------------------------------------------------------------------------|
| `brand`          | _string_ | An additional [card brand](#card-brands)                                                              |
| `funding`        | _string_ | An additional [funding type](#card-funding-types) of the {getProp(props, 'type')}                     |
| `authentication` | _string_ | An additional [authentication type](#authentication-types) required for this {getProp(props, 'type')} |
| `issuer`         | object   | Describes the country and issuing bank. See [Issuer](#issuer) for details                             |
